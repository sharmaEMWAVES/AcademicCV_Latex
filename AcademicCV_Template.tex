%CVSECTION 
\newcommand{\cvsection}[1]
{
	\begin{center}
		\Large\textcolor{sectcol}{\textbf{#1}}
	\end{center}
}

%CVSUBSECTION
\newcommand{\cvsubsection}[1]
{
\begin{center}
  \large\textcolor{subsectcol}{\textbf{#1}}\\[2pt]
\end{center}
}
% CVMETASECTION
\newcommand{\cvmetasection}[1]
{
\begin{center}
  \normalsize{#1}\\ [2pt] 
\end{center}
}

%-----------------------------------------------------------------------------
%	 CV EVENT
%----------------------------------------------------------------------------- 

% param 1:	event time i.e. 2022 or 2021-2022 etc.
% param 2:	Position i.e. Postdoc fellow/assistant professor/associate professor/professor
% param 3:	institution (where did you work / study)
% param 4:	Some itemize details
%
\newcommand{\cvevent}[4]
{
\begin{tabular*}{1\textwidth}{p{13.9cm}  x{3.4cm}}
	\textbf{#2} - \textcolor{bgcol}{#3} &   \vspace{1pt}\textcolor{sectcol}{#1}\end{tabular*}
  #4
\vspace{4pt}
}

%create a coloured arrow with title as cv meta section section
% param 1: meta section title
%
\newcommand{\metasection}[2]
{
\footnotesize{#2} \hspace*{\fill} \footnotesize{#1}\\[1pt]
}
